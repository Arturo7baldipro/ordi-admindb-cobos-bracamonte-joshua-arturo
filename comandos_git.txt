
git config --global user.name "Tu Nombre"
git config --global user.email "tu@email.com"

# Ver configuración
git config --list


INICIAR UN REPOSITORIO:
-----------------------
# Crear un nuevo repositorio logitcal
git init

# Clonar un repositorio existente
git clone https://github.com/usuario/repositorio.git


COMANDOS BÁSICOS DE COMMITS:
-----------------------------
Ver estado de archivos (modificados, sin seguimiento, etc.)
git status

Agregar archivos al staging area
git add nombre_archivo.txt          # Un archivo específico
git add .                            # Todos los archivos modificados
git add *.sql                        # Todos los archivos .sql

# Hacer un commit
git commit -m "Mensaje descriptivo del commit"

# Agregar y hacer commit en un solo paso (solo archivos ya rastreados)
git commit -am "Mensaje del commit"

# Ver historial de commits
git log                              # Completo
git log --oneline                    # Resumido
git log --graph --oneline --all      # Con gráfico de ramas


TRABAJAR CON RAMAS (BRANCHES):
------------------------------
# Ver ramas existentes
git branch

# Crear una nueva rama
git branch nombre-rama

# Cambiar a otra rama
git checkout nombre-rama

# Crear y cambiar a una nueva rama en un solo comando
git checkout -b nombre-rama

# Eliminar una rama
git branch -d nombre-rama            # Solo si está fusionada
git branch -D nombre-rama            # Forzar eliminación

# Renombrar rama actual
git branch -m nuevo-nombre


FUSIONAR RAMAS (MERGE):
-----------------------
# Fusionar una rama en la rama actual
git checkout main                    # Cambiar a la rama destino
git merge nombre-rama                # Fusionar la rama


TRABAJAR CON REPOSITORIOS REMOTOS:
----------------------------------
# Ver repositorios remotos configurados
git remote -v

# Agregar un repositorio remoto
git remote add origin https://github.com/usuario/repositorio.git

# Subir cambios al repositorio remoto
git push origin main                 # Subir rama main
git push origin nombre-rama          # Subir otra rama
git push -u origin main              # Primera vez (establece upstream)

# Descargar cambios del repositorio remoto
git pull origin main                 # Descargar y fusionar
git fetch origin                     # Solo descargar sin fusionar


DESHACER CAMBIOS:
-----------------
# Descartar cambios en un archivo (antes de add)
git checkout -- nombre_archivo.txt

# Quitar archivo del staging area (después de add)
git reset nombre_archivo.txt

# Deshacer último commit (mantiene cambios)
git reset --soft HEAD~1

# Deshacer último commit (descarta cambios)
git reset --hard HEAD~1


VER DIFERENCIAS:
----------------
# Ver cambios no agregados
git diff

# Ver cambios en staging area
git diff --staged

# Ver diferencias entre ramas
git diff rama1 rama2


OTROS COMANDOS ÚTILES:
----------------------
# Guardar cambios temporalmente sin commit
git stash
git stash pop                        # Recuperar cambios guardados
git stash list                       # Ver lista de stashes

# Ver quién modificó cada línea de un archivo
git blame nombre_archivo.txt

# Crear etiquetas (tags)
git tag v1.0.0
git push origin v1.0.0


FLUJO DE TRABAJO TÍPICO:
------------------------
1. git pull origin main              # Actualizar desde remoto
2. git checkout -b feature/nueva-funcionalidad
3. # ... hacer cambios en archivos ...
4. git add .
5. git commit -m "Agregar nueva funcionalidad"
6. git push origin feature/nueva-funcionalidad
7. # Crear Pull Request en GitHub
8. git checkout main
9. git merge feature/nueva-funcionalidad
10. git push origin main


EJEMPLO PRÁCTICO:
-----------------
# Iniciar proyecto
git init
git add .
git commit -m "Primer commit: estructura inicial"

# Crear repositorio en GitHub y conectar
git remote add origin https://github.com/usuario/mi-proyecto.git
git push -u origin main

# Crear rama para nueva característica
git checkout -b feature/agregar-productos
# ... modificar archivos ...
git add .
git commit -m "Agregar tabla de productos"
git push origin feature/agregar-productos

# Volver a main y fusionar
git checkout main
git merge feature/agregar-productos
git push origin main


RESOLVER CONFLICTOS:
--------------------
1. git pull origin main              # Si hay conflicto, Git lo marcará
2. # Abrir archivos en conflicto y editarlos manualmente
3. # Buscar marcadores: <<<<<<< HEAD, =======, >>>>>>>
4. git add archivo_resuelto.txt
5. git commit -m "Resolver conflicto de fusión"
6. git push origin main


COMANDOS DE AYUDA:
------------------
git help                             # Ayuda general
git help comando                     # Ayuda de comando específico
git comando --help                   # Igual que anterior
